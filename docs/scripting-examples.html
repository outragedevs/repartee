Scripting — Examples

Note: Scripts run in a sandboxed Lua 5.4 environment. The os, io, loadfile, dofile, and package globals are removed. The math, string, table, coroutine, and utf8 libraries are available. Use api.log() for debug output and api.ui.print() for UI messages.

Practical Lua script examples for Repartee. Save these as .lua files in ~/.repartee/scripts/.

Auto-greet on join

Greet users when they join a channel:

meta = {
    name = "greet",
    version = "1.0",
    description = "Welcome joiners",
}

function setup(api)
    api.on("irc.join", function(ev)
        local our_nick = api.store.our_nick()
        if ev.nick ~= our_nick then
            api.irc.say(ev.channel, "Welcome, " .. ev.nick .. "!")
        end
    end)
end

Dice game

Respond to !dice in channels with random rolls:

meta = {
    name = "dice",
    version = "1.0",
    description = "!dice game -- roll dice in channel",
}

function setup(api)
    api.on("irc.privmsg", function(ev)
        local cmd, rest = (ev.message or ""):match("^(!dice)%s*(.*)")
        if not cmd then return end

        local count, sides = 1, 6
        if rest ~= "" then
            local c, s = rest:match("^(%d+)d(%d+)$")
            if c then count, sides = tonumber(c), tonumber(s)
            else
                local s2 = rest:match("^d(%d+)$")
                if s2 then sides = tonumber(s2) end
            end
        end

        count = math.max(1, math.min(count, 20))
        sides = math.max(2, math.min(sides, 1000))

        local total = 0
        for i = 1, count do total = total + math.random(1, sides) end
        api.irc.say(ev.channel, ev.nick .. " rolls " .. count .. "d" .. sides .. ": " .. total)
    end)
end

URL logger

Log all URLs posted to channels, with a /urls command to review them:

meta = {
    name = "urllogger",
    version = "1.0",
    description = "Log URLs from messages",
}

function setup(api)
    local urls = {}

    api.on("irc.privmsg", function(ev)
        for url in ev.message:gmatch("https?://[%w%.%-/%%?&=_#]+") do
            table.insert(urls, {
                nick = ev.nick,
                channel = ev.target,
                url = url,
            })
            api.log("URL: " .. url .. " from " .. ev.nick)
        end
    end)

    api.command("urls", {
        handler = function(args)
            local count = tonumber(args[1]) or 10
            local start = math.max(1, #urls - count + 1)
            for i = start, #urls do
                local u = urls[i]
                api.ui.print(u.nick .. " > " .. u.url)
            end
        end,
        description = "Show recent URLs",
        usage = "/urls [count]",
    })
end

Highlight monitor

Print a notification when your nick is mentioned:

meta = {
    name = "hilight",
    version = "1.0",
    description = "Monitor highlighted messages",
}

function setup(api)
    api.on("irc.privmsg", function(ev)
        local my_nick = api.store.our_nick()
        if my_nick and ev.message:lower():find(my_nick:lower(), 1, true) then
            local msg = "[" .. ev.target .. "] <" .. ev.nick .. "> " .. ev.message
            api.ui.print(msg)
        end
    end)
end

Custom slap command

The classic IRC /slap command:

meta = {
    name = "slap",
    version = "1.0",
    description = "Slap someone with a large trout",
}

function setup(api)
    local items = {
        "a large trout",
        "a mass of wet noodles",
        "a mass-produced plastic toy",
        "a mass of jello",
        "a mass of cotton candy",
    }

    api.command("slap", {
        handler = function(args)
            local target = args[1]
            if not target then
                api.ui.print("Usage: /slap ")
                return
            end
            local item = items[math.random(#items)]
            local buf = api.store.active_buffer()
            if buf then
                api.irc.action(buf, "slaps " .. target .. " around a bit with " .. item)
            end
        end,
        description = "Slap someone with a random object",
        usage = "/slap ",
    })
end

Spam filter

Block messages matching patterns (runs at high priority to suppress before display):

meta = {
    name = "spamfilter",
    version = "1.0",
    description = "Filter spam messages",
}

function setup(api)
    local patterns = {
        "buy cheap",
        "free bitcoins",
        "click here now",
    }

    -- High priority so we run before default handlers
    api.on("irc.privmsg", function(ev)
        local lower = ev.message:lower()
        for _, pattern in ipairs(patterns) do
            if lower:find(pattern, 1, true) then
                api.log("Blocked spam from " .. ev.nick .. ": " .. ev.message)
                return true  -- suppress the event
            end
        end
    end, api.PRIORITY_HIGH)
end

Reminder timer

Set a reminder with /remind <minutes> <message>:

meta = {
    name = "reminder",
    version = "1.0",
    description = "Set reminders",
}

function setup(api)
    api.command("remind", {
        handler = function(args)
            local minutes = tonumber(args[1]) or 5
            local message = table.concat(args, " ", 2)
            if message == "" then message = "Time's up!" end

            api.timeout(minutes * 60 * 1000, function()
                api.ui.print("Reminder: " .. message)
            end)
            api.ui.print("Reminder set for " .. minutes .. " minutes")
        end,
        description = "Set a reminder",
        usage = "/remind  ",
    })
end

Custom command example

Register a command that shows connection info:

meta = {
    name = "conninfo",
    version = "1.0",
    description = "Show connection details",
}

function setup(api)
    api.command("conninfo", {
        handler = function(args)
            local conns = api.store.connections()
            if #conns == 0 then
                api.ui.print("No connections")
                return
            end
            for _, id in ipairs(conns) do
                local info = api.store.connection_info(id)
                if info then
                    api.ui.print(id .. ": " .. (info.server or "unknown") .. " as " .. (info.nick or "?"))
                end
            end
        end,
        description = "Show all connection details",
        usage = "/conninfo",
    })
end
